Archive - Post Action

xcodebuild -scheme <name> clean archive
<PostActions>
	<ExecutionAction 
		ActionType = "Xcode.IDEStandardExecutionActionsCore.ExecutionActionType.ShellScriptAction">
		<ActionContent 
				title = "Run Script" 
				scriptText = "echo CI_ARCHIVE_PATH="$ARCHIVE_PATH"">
		</ActionContent>
	</ExecutionAction>
</PostActions>
